Verilog Code for half subtractor

Gate-level Modeling 
                    
    module hsa(
        input a, 
        input b, 
        output d, 
        output borrow
        );
        
        wire n;
        xor(d,a,b);
        not(n,a);
        and(borrow,n,b);
    
    endmodule
Behavioral Modeling 
                      
    module hsa(
        input a,
        input b,
        output d,
        output borrow
    );
        assign d = a ^ b;
        assign borrow = (!a) && b;

    endmodule
